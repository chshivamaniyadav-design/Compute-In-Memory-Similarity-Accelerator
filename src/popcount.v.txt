`timescale 1ns / 1ps

module popcount #
(
    parameter WIDTH = 128
)
(
    input  [WIDTH-1:0] data_in,
    output reg [$clog2(WIDTH+1)-1:0] count
);

integer i;

always @(*) 
begin
    count = 0;

    for (i = 0; i < WIDTH; i = i + 1)
    begin
        count = count + data_in[i] + data_in[i];
    end
end

endmodule
